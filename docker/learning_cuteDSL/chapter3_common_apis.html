MMA (Matrix Multiply) APIs

# Create MMA atom
mma_atom = cute.MmaAtom.F16F16F16_TN

# Create tiled MMA (thread-level)
tiled_mma = cute.make_tiled_mma(
    mma_atom,
    cute.LayoutType.RowMajor,
    cute.LayoutType.ColMajor
)

# Get thread slice
thr_mma = tiled_mma.get_slice(thread_id)

# Partition tensors for MMA
tCgA = thr_mma.partition_A(gA)  # A partition
tCgB = thr_mma.partition_B(gB)  # B partition
tCgC = thr_mma.partition_C(gC)  # C partition

# Create accumulator fragment
acc = tiled_mma.make_fragment_C(shape)

# Perform GEMM
cute.gemm(tiled_mma, acc, tCrA, tCrB, acc)

Copy APIs

# Create copy atom
copy_atom = cute.make_copy_atom(
    cute.nvgpu.CopyUniversalOp(),
    cutlass.Float16
)

# Create tiled copy
tiled_copy = cute.make_tiled_copy(copy_atom)

# Get thread slice
thr_copy = tiled_copy.get_slice(thread_id)

# Partition for copy
tCsS = thr_copy.partition_S(smem)  # Source
tCsD = thr_copy.partition_D(gmem)   # Destination

# Perform copy
cute.copy(tiled_copy, tCsS, tCsD)

TMA (Tensor Memory Access) - Hopper/Blackwell

import cute.nvgpu as nvgpu

# Create TMA atom for A matrix
tma_atom_a, tma_tensor_a = nvgpu.make_tiled_tma_atom_A(
    cluster_shape,     # Cluster layout
    tensor_a,          # Global tensor
    smem_layout,       # Shared memory layout
    tile_shape,        # Tile shape
    tiled_mma,         # Tiled MMA
    cluster_layout
)

# Prefetch descriptor
cpasync.prefetch_descriptor(tma_atom_a)

# TMA load
cute.copy(
    tma_atom_a,
    gA_tile,      # Global address
    sA_tile,     # Shared memory address
    tma_barrier
)

Pipeline APIs (for async operations)

import cutlass.pipeline as pipeline

# Create pipeline
pipeline = pipeline.PipelineTmaUmma.create(
    barrier_storage=storage.barrier_ptr,
    num_stages=3,
    producer_group=producer_group,
    consumer_group=consumer_group,
    cta_layout=cluster_layout
)

# Producer operations
handle = pipeline.producer_acquire()
pipeline.producer_commit(handle)

# Consumer operations  
pipeline.consumer_wait(handle)
pipeline.consumer_release()

Arch APIs

import cute.arch as arch

# Thread/block indices
tid = arch.thread_idx()
bid = arch.block_idx()

# Warp info
warp_id = arch.warp_idx()
warp_size = arch.warp_size()

# Cluster info
cta_in_cluster = arch.block_idx_in_cluster()

# Synchronization
arch.fence_proxy("async.shared", space="cta")
arch.sync()  # Full barrier

# Memory operations
arch.warpgroup_arrive()
arch.warpgroup_commit()
