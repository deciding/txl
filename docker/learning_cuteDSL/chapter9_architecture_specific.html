SM80 (Ampere)

# Basic MMA operations
tiled_mma = cute.make_tiled_mma(
    cute.MmaAtom.F16F16F16_TN,
    cute.LayoutType.RowMajor,
    cute.LayoutType.ColMajor
)

# No TMA - use regular copy
cute.copy(copy_atom, gmem, smem)

SM90 (Hopper)

Key Features:

  • TMA: Hardware-accelerated memory copies
  • Clusters: Cooperating CTAs
  • Async Copy: Overlapped loads
  • TMEM: Large accumulator storage
# TMA Load
tma_atom, tensor = cute.nvgpu.make_tiled_tma_atom_A(...)

# Cluster multicast
mask = cpasync.create_tma_multicast_mask(
    cluster_layout, coord, mcast_mode=2
)
cute.copy(tma_atom, src, dst, mcast_mask=mask)

SM100 (Blackwell)

New Features:

  • 2-CTA MMA: Two CTAs per MMA tile
  • Enhanced TMEM: More capacity
  • New TMA features: Better multicast
# 2-CTA MMA (use two CTAs for one MMA tile)
cta_group = tcgen05.CtaGroup.TWO  # vs tcgen05.CtaGroup.ONE
tiled_mma = sm100_utils.make_trivial_tiled_mma(
    dtype, major_a, major_b, acc_dtype,
    cta_group,  # Use TWO for 2-CTA
    tile_shape
)

Detecting Architecture

import torch

def get_sm_version():
    props = torch.cuda.get_device_properties()
    return props.major * 10 + props.minor

# Usage
if get_sm_version() >= 100:
    # Use Blackwell features
elif get_sm_version() >= 90:
    # Use Hopper features
else:
    # Use Ampere features
